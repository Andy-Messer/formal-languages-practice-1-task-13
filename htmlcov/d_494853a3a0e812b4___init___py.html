24def solve(istream=sys.stdin, ostream=sys.stdout): 

25 """! Solution for the task 

26 Корректность: воспользуемся тем, что любая регулярка задаёт НКА. 

27 А значит если подслово может быть прочитано НКА, 

28 то оно принадлежит установленному языку. 

29 Более точно, переберем все подстроки данной нам строки и попытаемся их прочитать, 

30 длина наибольшей прочитанной строки будет ответом. 

31 Верность построения НКА доказывается самим алгоритмом построения. 

32 """ 

33 logging.info(f'Redirecting input and output streams') 

34 

35 cp_stdin = sys.stdin 

36 cp_stout = sys.stdout 

37 if isinstance(istream, type(io.StringIO())): 

38 sys.stdin = istream 

39 if isinstance(ostream, type(io.StringIO())): 

40 sys.stdout = ostream 

41 logging.info(f'Done...') 

42 

43 try: 

44 input_split = input().split() 

45 string = input_split[-1] 

46 reg = "".join(input_split[:len(input_split) - 1]) 

47 logging.info(f'Submitted to the entrance: "{reg}", "{string}"') 

48 try: 

49 logging.info(f'Start building NFA...') 

50 try: 

51 nfa_global = NFA(reg) 

52 except IndexError as e: 

53 return e 

54 logging.debug(f'Dumping NFA... (to see dumpfile go to "./logs/dump.log" or "./snapshots"') 

55 nfa_global._make_dump() 

56 

57 logging.info(f'Start solving the task.') 

58 ans = 0 

59 for i in range(len(string)): 

60 for j in range(i + 1, len(string) + 1): 

61 logging.info(f'------------------------------------------------') 

62 

63 substring = string[i:j] 

64 logging.info(f'"{substring}" has been passed to NFA for reading') 

65 

66 if nfa_global.read(substring): 

67 logging.info(f'the Line was read!') 

68 

69 logging.info(f'Checking the length of substring') 

70 if len(substring) > ans: 

71 logging.info(f'New answer!!! {len(substring)}') 

72 

73 ans = max(len(substring), ans) 

74 

75 logging.info(f'Checked.') 

76 else: 

77 logging.info(f'the Line wasn\'t read') 

78 print(ans) 

79 if isinstance(istream, type(io.StringIO())): 

80 sys.stdin = cp_stdin 

81 if isinstance(ostream, type(io.StringIO())): 

82 sys.stdout = cp_stout 

83 

84 logging.info('Exiting from "solve"') 

85 return ans 

86 

87 except IndexError as e: 

88 logging.error('failed to build the NFA')
